cmake_minimum_required(VERSION 3.1.3)
set (CMAKE_CXX_STANDARD 11)
#Find libiio and link it to the targets generated by this project

if (NOT WIN32)
	find_library(PTHREAD_LIBRARIES pthread)
endif()

if (LIBM2K_VERSION)			# in-source build
	set(PROJECT_VERSION ${LIBM2K_VERSION})
else()					# standalone build
	set(PROJECT_VERSION "1.0.0")
	find_package(libm2k REQUIRED)
endif()
add_library(libm2k_lv SHARED "${CMAKE_CURRENT_SOURCE_DIR}/libm2k_lv.cpp" 
				"${CMAKE_CURRENT_SOURCE_DIR}/libm2k_lv.h")

target_link_libraries(libm2k_lv PRIVATE libm2k::libm2k)
